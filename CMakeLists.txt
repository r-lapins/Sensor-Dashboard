cmake_minimum_required(VERSION 3.19)
project(SensorDashboard LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 6.5 REQUIRED COMPONENTS Core Widgets)

    mainwindow.cpp

    mainwindow.ui
)

    PRIVATE
        Qt::Core
        Qt::Widgets
